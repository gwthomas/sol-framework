92  {
93  if (y == 0.0) return 0.0;
94  else if (y > 0.0) return M_PI_2;
95  else return 3.0 * M_PI_2;
96  }
97  else if (x > 0.0)

99  if (y == 0.0) return 0.0;

102  }
103  else

106  else return M_PI + atanf(y / x);

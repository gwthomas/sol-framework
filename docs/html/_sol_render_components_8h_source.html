154  bool VertexAttribArray<T, SIZE, GL_TYPE, NORMALIZED, LOCATION>::submit()
155  {

158  return false;

162  {

168  }
169 
170  glVertexAttribPointer(_location, SIZE, GL_TYPE, NORMALIZED, 0, 0);
171  }
172  else

176  }
